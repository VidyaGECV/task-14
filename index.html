3. Fibonacci Series

        function fibonacci(n) {
        let fib = [0, 1];
        for (let i = 2; i < n; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }
        return fib.slice(0, n);
        }

4. Sum of Multiples of 3 and 5 Below 1000

        function sumMultiplesOf3And5(limit) {
        let sum = 0;
        for (let i = 0; i < limit; i++) {
            if (i % 3 === 0 || i % 5 === 0) {
            sum += i;
            }
        }
        return sum;
        }
